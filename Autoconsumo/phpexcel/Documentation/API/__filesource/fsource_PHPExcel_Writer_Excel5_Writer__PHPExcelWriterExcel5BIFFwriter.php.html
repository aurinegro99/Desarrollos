  35. /**
    36. 

  36. * Class for writing Excel BIFF records.
    37. 

  37. *
    38. 

  38. * From "MICROSOFT EXCEL BINARY FILE FORMAT" by Mark O'Brien (Microsoft Corporation):
    39. 

  39. *
    40. 

  40. * BIFF (BInary File Format) is the file format in which Excel documents are
    41. 

  41. * saved on disk.  A BIFF file is a complete description of an Excel document.
    42. 

  42. * BIFF files consist of sequences of variable-length records. There are many
    43. 

  43. * different types of BIFF records.  For example, one record type describes a
    44. 

  44. * formula entered into a cell; one describes the size and location of a
    45. 

  45. * window into a document; another describes a picture format.
    46. 

  46. *
    47. 

  47. * @author   Xavier Noguer <xnoguer@php.net>
    48. 

  48. * @category FileFormats
    49. 

  49. * @package  PHPExcel_Writer_Excel5_Writer
    50. 

  50. */

  61.     * The byte order of this architecture. 0 => little endian, 1 => big endian

  62.     * @var integer

  74.     * @var integer

  75.     */
    76.

  79.     * The maximun length for a BIFF record. See _addContinue()

  89.     */
    90. 

  90.     function PHPExcel_Writer_Excel5_BIFFwriter()
    91. 

  91.     {
    92. 

  92.         $this->_byte_order = '';
    93. 

  93.         $this->_data       = '';
    94. 

  94.         $this->_datasize   = 0;
    95. 

  95.         $this->_limit      = 2080;
    96. 

  96.         // Set the byte order
    97. 

  97.         $this->_setByteOrder();
    98. 

  98.     }
    99.

    100. 

  100.     /**
    101. 

  101.     * Determine the byte order and store it as class data to avoid
    102. 

  102.     * recalculating it for each call to new().
    103. 

  103.     *
    104. 

  104.     * @access private
    105. 

  105.     */
    106. 

  106.     function _setByteOrder()
    107. 

  107.     {
    108. 

  108.         // Check if "pack" gives the required IEEE 64bit float
    109. 

  109.         $teststr = pack("d", 1.2345);
    110. 

  110.         $number  = pack("C8", 0x8D, 0x97, 0x6E, 0x12, 0x83, 0xC0, 0xF3, 0x3F);
    111. 

  111.         if ($number == $teststr) {
    112. 

  112.             $byte_order = 0;    // Little Endian
    113. 

  113.         } elseif ($number == strrev($teststr)){
    114. 

  114.             $byte_order = 1;    // Big Endian
    115. 

  115.         } else {
    116. 

  116.             // Give up. I'll fix this in a later version.
    117. 

  117.             throw new Exception("Required floating point format ".
    118. 

  118.                                      "not supported on this platform.");
    119. 

  119.         }
    120. 

  120.         $this->_byte_order = $byte_order;
    121. 

  121.     }
    122.

  123.     /**
    124. 

  124.     * General storage function
    125. 

  125.     *
    126. 

  126.     * @param string $data binary data to prepend
    127. 

  127.     * @access private

  129.     function _prepend($data)
    130. 

  130.     {
    131. 

  131.         if (strlen($data) > $this->_limit) {
    132. 

  132.             $data = $this->_addContinue($data);
    133. 

  133.         }
    134. 

  134.         $this->_data      = $data.$this->_data;
    135. 

  135.         $this->_datasize += strlen($data);
    136. 

  136.     }

    146.

    148. 

  148.         }

  149.         $this->_data      = $this->_data.$data;

  151.     }

  153.     /**
    154. 

  154.     * Writes Excel BOF record to indicate the beginning of a stream or
    155. 

  155.     * sub-stream in the BIFF file.
    156. 

  156.     *
    157. 

  157.     * @param  integer $type Type of BIFF file to write: 0x0005 Workbook,
    158. 

  158.     *                        0x0010 Worksheet.
    159. 

  159.     * @access private
    160. 

  160.     */
    161. 

  161.     function _storeBof($type)
    162. 

  162.     {
    163. 

  163.         $record  = 0x0809;        // Record identifier
    164. 

  164.  
    165. 

  165.         // According to the SDK $build and $year should be set to zero.
    166. 

  166.         // However, this throws a warning in Excel 5. So, use magic numbers.
    167. 

  167.         if ($this->_BIFF_version == 0x0500) {
    168. 

  168.             $length  = 0x0008;
    169. 

  169.             $unknown = '';
    170. 

  170.             $build   = 0x096C;
    171. 

  171.             $year    = 0x07C9;
    172. 

  172.         } elseif ($this->_BIFF_version == 0x0600) {
    173. 

  173.             $length  = 0x0010;
    174. 

  174.             $unknown = pack("VV", 0x00000041, 0x00000006); //unknown last 8 bytes for BIFF8
    175. 

  175.             $build   = 0x0DBB;
    176. 

  176.             $year    = 0x07CC;
    177. 

  177.         }
    178. 

  178.         $version = $this->_BIFF_version;
    179. 

  179.  
    180. 

  180.         $header  = pack("vv",   $record, $length);
    181. 

  181.         $data    = pack("vvvv", $version, $type, $build, $year);
    182. 

  182.         $this->_prepend($header . $data . $unknown);
    183. 

  183.     }
    184. 

  184.  
    185. 

  185.     /**
    186. 

  186.     * Writes Excel EOF record to indicate the end of a BIFF stream.
    187. 

  187.     *
    188. 

  188.     * @access private
    189. 

  189.     */
    190. 

  190.     function _storeEof()
    191. 

  191.     {
    192. 

  192.         $record    = 0x000A;   // Record identifier
    193. 

  193.         $length    = 0x0000;   // Number of bytes to follow
    194. 

  194.         $header    = pack("vv", $record, $length);
    195. 

  195.         $this->_append($header);
    196. 

  196.     }
    197. 

  197.  
    198. 

  198.     /**
    199. 

  199.     * Excel limits the size of BIFF records. In Excel 5 the limit is 2084 bytes. In
    200. 

  200.     * Excel 97 the limit is 8228 bytes. Records that are longer than these limits
    201. 

  201.     * must be split up into CONTINUE blocks.
    202. 

  202.     *
    203. 

  203.     * This function takes a long BIFF record and inserts CONTINUE records as
    204. 

  204.     * necessary.
    205. 

  205.     *
    206. 

  206.     * @param  string  $data The original binary data to be written
    207. 

  207.     * @return string        A very convenient string of continue blocks
    208. 

  208.     * @access private
    209. 

  209.     */
    210. 

  210.     function _addContinue($data)
    211. 

  211.     {
    212. 

  212.         $limit  = $this->_limit;
    213. 

  213.         $record = 0x003C;         // Record identifier
    214. 

  214.  
    215. 

  215.         // The first 2080/8224 bytes remain intact. However, we have to change
    216. 

  216.         // the length field of the record.
    217. 

  217.         $tmp = substr($data, 0, 2).pack("v", $limit-4).substr($data, 4, $limit - 4);
    218. 

  218.  
    219. 

  219.         $header = pack("vv", $record, $limit);  // Headers for continue records
    220. 

  220.  
    221. 

  221.         // Retrieve chunks of 2080/8224 bytes +4 for the header.
    222. 

  222.         $data_length = strlen($data);
    223. 

  223.         for ($i = $limit; $i <  ($data_length - $limit); $i += $limit) {
    224. 

  224.             $tmp .= $header;
    225. 

  225.             $tmp .= substr($data, $i, $limit);
    226. 

  226.         }
    227. 

  227.  
    228. 

  228.         // Retrieve the last chunk of data
    229. 

  229.         $header  = pack("vv", $record, strlen($data) - $i);
    230. 

  230.         $tmp    .= $header;
    231. 

  231.         $tmp    .= substr($data, $i, strlen($data) - $i);
    232. 

  232.  
    233. 

  233.         return $tmp;
    234. 

  234.     }
