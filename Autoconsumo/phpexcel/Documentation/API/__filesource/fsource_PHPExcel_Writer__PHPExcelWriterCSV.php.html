  111.     public function save($pFilename = null) {
    112. 

  112.         // Fetch sheet
    113. 

  113.         $sheet = $this->_phpExcel->getSheet($this->_sheetIndex);
    114. 

  114.         
    115. 

  115.         // Open file
    116. 

  116.         $fileHandle = fopen($pFilename, 'w');
    117. 

  117.         if ($fileHandle === false) {
    118. 

  118.             throw new Exception("Could not open file $pFilename for writing.");
    119. 

  119.         }
    120. 

  120.         
    121. 

  121.         // Get cell collection
    122. 

  122.         $cellCollection = $sheet->getCellCollection();
    123. 

  123.         
    124. 

  124.         // Get column count
    125. 

  125.         $colCount = PHPExcel_Cell::columnIndexFromString($sheet->getHighestColumn());
    126. 

  126.         
    127. 

  127.         // Loop trough cells
    128. 

  128.         $currentRow = -1;
    129. 

  129.         $rowData = array();
    130. 

  130.         foreach ($cellCollection as $cell) {                    
    131. 

  131.             if ($currentRow != $cell->getRow()) {
    132. 

  132.                 // End previous row?
    133. 

  133.                 if ($currentRow != -1) {
    134. 

  134.                     $this->_writeLine($fileHandle, $rowData);
    135. 

  135.                 }
    136. 

  136.  
    137. 

  137.                 // Set current row
    138. 

  138.                 $currentRow = $cell->getRow();
    139. 

  139.             
    140. 

  140.                 // Start a new row
    141. 

  141.                 $rowData = array();
    142. 

  142.                 for ($i = 0; $i < $colCount; $i++) {
    143. 

  143.                     $rowData[$i] = '';
    144. 

  144.                 }
    145. 

  145.             }
    146. 

  146.                     
    147. 

  147.             // Copy cell
    148. 

  148.             $column = PHPExcel_Cell::columnIndexFromString($cell->getColumn()) - 1;
    149. 

  149.             if ($cell->getValue() instanceof PHPExcel_RichText) {
    150. 

  150.                 $rowData[$column] = $cell->getValue()->getPlainText();
    151. 

  151.             } else {
    152. 

  152.                 if ($this->_preCalculateFormulas) {
    153. 

  153.                     $rowData[$column] = PHPExcel_Style_NumberFormat::toFormattedString(
    154. 

  154.                         $cell->getCalculatedValue(),
    155. 

  155.                         $sheet->getstyle( $cell->getCoordinate() )->getNumberFormat()->getFormatCode()
    156. 

  156.                     );
    157. 

  157.                 } else {
    158. 

  158.                     $rowData[$column] = PHPExcel_Style_NumberFormat::ToFormattedString(
    159. 

  159.                         $cell->getValue(),
    160. 

  160.                         $sheet->getstyle( $cell->getCoordinate() )->getNumberFormat()->getFormatCode()
    161. 

  161.                     );
    162. 

  162.                 }
    163. 

  163.             }
    164. 

  164.         }
    165. 

  165.         
    166. 

  166.         // End last row?
    167. 

  167.         if ($currentRow != -1) {
    168. 

  168.             $this->_writeLine($fileHandle, $rowData);
    169. 

  169.         }
    170. 

  170.                 
    171. 

  171.         // Close file
    172. 

  172.         fclose($fileHandle);
    173. 

  173.     }

  257.     private function _writeLine($pFileHandle = null, $pValues = null) {
    258. 

  258.         if (!is_null($pFileHandle) && is_array($pValues)) {
    259. 

  259.             // No leading delimiter
    260. 

  260.             $writeDelimiter = false;
    261. 

  261.             
    262. 

  262.             // Build the line
    263. 

  263.             $line = '';
    264. 

  264.             
    265. 

  265.             foreach ($pValues as $element) {
    266. 

  266.                 // Escape enclosures
    267. 

  267.                 $element = str_replace($this->_enclosure, "\\" . $this->_enclosure, $element);
    268. 

  268.  
    269. 

  269.                 // Add delimiter
    270. 

  270.                 if ($writeDelimiter) {
    271. 

  271.                     $line .= $this->_delimiter;
    272. 

  272.                 } else {
    273. 

  273.                     $writeDelimiter = true;
    274. 

  274.                 }
    275. 

  275.                 
    276. 

  276.                 // Add enclosed string
    277. 

  277.                 $line .= $this->_enclosure . $element . $this->_enclosure;
    278. 

  278.             }
    279. 

  279.             
    280. 

  280.             // Add line ending
    281. 

  281.             $line .= $this->_lineEnding;
    282. 

  282.             
    283. 

  283.             // Write to file
    284. 

  284.             fwrite($pFileHandle, $line);
    285. 

  285.         } else {
    286. 

  286.             throw new Exception("Invalid parameters passed.");
    287. 

  287.         }
    288. 

  288.     }
