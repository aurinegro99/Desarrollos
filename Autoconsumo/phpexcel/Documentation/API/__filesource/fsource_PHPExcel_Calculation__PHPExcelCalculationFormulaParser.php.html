  161.     private function _parseToTokens() {
    162. 

  162.         // No attempt is made to verify formulas; assumes formulas are derived from Excel, where 
    163. 

  163.         // they can only exist if valid; stack overflows/underflows sunk as nulls without exceptions.
    164. 

  164.  
    165. 

  165.         // Check if the formula has a valid starting =
    166. 

  166.         if (strlen($this->_formula) < 2 || substr($this->_formula, 0, 1) != '=') return;
    167. 

  167.               
    168. 

  168.         // Helper variables
    169. 

  169.         $tokens1     = array();
    170. 

  170.         $tokens2     = array();
    171. 

  171.         $stack         = array();
    172. 

  172.         $inString    = false;
    173. 

  173.         $inPath     = false;
    174. 

  174.         $inRange     = false;
    175. 

  175.         $inError     = false;
    176. 

  176.               
    177. 

  177.         $index    = 1;
    178. 

  178.         $value    = '';
    179. 

  179.               
    180. 

  180.         $ERRORS             = array("#NULL!", "#DIV/0!", "#VALUE!", "#REF!", "#NAME?", "#NUM!", "#N/A");
    181. 

  181.         $COMPARATORS_MULTI     = array(">=", "<=", "<>");
    182. 

  182.         
    183. 

  183.         $token             = null;
    184. 

  184.         $previousToken    = null;
    185. 

  185.         $nextToken        = null;
    186. 

  186.         
    187. 

  187.         while ($index < strlen($this->_formula)) {
    188. 

  188.             // state-dependent character evaluation (order is important)
    189. 

  189.                     
    190. 

  190.             // double-quoted strings
    191. 

  191.             // embeds are doubled
    192. 

  192.             // end marks token
    193. 

  193.             if ($inString) {    
    194. 

  194.                 if (substr($this->_formula, $index, 1) == PHPExcel_Calculation_FormulaParser::QUOTE_DOUBLE) {
    195. 

  195.                     if ((($index + 2) <= strlen($this->_formula)) && (substr($this->_formula, $index + 1, 1) == PHPExcel_Calculation_FormulaParser::QUOTE_DOUBLE)) {
    196. 

  196.                         $value .= PHPExcel_Calculation_FormulaParser::QUOTE_DOUBLE;
    197. 

  197.                         $index++;
    198. 

  198.                     } else {
    199. 

  199.                         $inString = false;
    200. 

  200.                         array_push(
    201. 

  201.                             $tokens1,
    202. 

  202.                             new PHPExcel_Calculation_FormulaToken($value, PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERAND, PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_TEXT)
    203. 

  203.                         );
    204. 

  204.                         $value = "";
    205. 

  205.                     }      
    206. 

  206.                 } else {
    207. 

  207.                     $value .= substr($this->_formula, $index, 1);
    208. 

  208.                 }
    209. 

  209.                 $index++;
    210. 

  210.                 continue;
    211. 

  211.             }
    212. 

  212.             
    213. 

  213.             // single-quoted strings (links)
    214. 

  214.             // embeds are double
    215. 

  215.             // end does not mark a token
    216. 

  216.             if ($inPath) {
    217. 

  217.                 if (substr($this->_formula, $index, 1) == PHPExcel_Calculation_FormulaParser::QUOTE_SINGLE) {
    218. 

  218.                     if ((($index + 2) <= strlen($this->_formula)) && (substr($this->_formula, $index + 1, 1) == PHPExcel_Calculation_FormulaParser::QUOTE_SINGLE)) {
    219. 

  219.                         $value .= PHPExcel_Calculation_FormulaParser::QUOTE_SINGLE;
    220. 

  220.                         $index++;
    221. 

  221.                     } else {
    222. 

  222.                         $inPath = false;
    223. 

  223.                     }
    224. 

  224.                 } else {
    225. 

  225.                     $value .= substr($this->_formula, $index, 1);
    226. 

  226.                 }
    227. 

  227.                 $index++;
    228. 

  228.                 continue; 
    229. 

  229.             }
    230. 

  230.             
    231. 

  231.             // bracked strings (R1C1 range index or linked workbook name)
    232. 

  232.             // no embeds (changed to "()" by Excel)
    233. 

  233.             // end does not mark a token
    234. 

  234.             if ($inRange) {
    235. 

  235.                 if (substr($this->_formula, $index, 1) == PHPExcel_Calculation_FormulaParser::BRACKET_CLOSE) {
    236. 

  236.                     $inRange = false;
    237. 

  237.                 }
    238. 

  238.                 $value .= substr($this->_formula, $index, 1);
    239. 

  239.                 $index++;
    240. 

  240.                 continue;
    241. 

  241.             }
    242. 

  242.             
    243. 

  243.             // error values
    244. 

  244.             // end marks a token, determined from absolute list of values
    245. 

  245.             if ($inError) {
    246. 

  246.                 $value .= substr($this->_formula, $index, 1);
    247. 

  247.                 $index++;
    248. 

  248.                 if (in_array($value, $ERRORS)) {
    249. 

  249.                     $inError = false;
    250. 

  250.                     array_push(
    251. 

  251.                         $tokens1,
    252. 

  252.                         new PHPExcel_Calculation_FormulaToken($value, PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERAND, PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_ERROR)
    253. 

  253.                     );
    254. 

  254.                     $value = "";
    255. 

  255.                 }
    256. 

  256.                 continue;
    257. 

  257.             }
    258. 

  258.             
    259. 

  259.             // scientific notation check
    260. 

  260.             if (strpos(PHPExcel_Calculation_FormulaParser::OPERATORS_SN, substr($this->_formula, $index, 1)) !== false) {
    261. 

  261.                 if (strlen($value) > 1) {
    262. 

  262.                     if (preg_match("/^[1-9]{1}(\.[0-9]+)?E{1}$/", substr($this->_formula, $index, 1)) != 0) {
    263. 

  263.                         $value .= substr($this->_formula, $index, 1);
    264. 

  264.                         $index++;
    265. 

  265.                         continue;
    266. 

  266.                     }
    267. 

  267.                 }
    268. 

  268.             }
    269. 

  269.             
    270. 

  270.             // independent character evaluation (order not important)
    271. 

  271.  
    272. 

  272.             // establish state-dependent character evaluations
    273. 

  273.             if (substr($this->_formula, $index, 1) == PHPExcel_Calculation_FormulaParser::QUOTE_DOUBLE) {  
    274. 

  274.                 if (strlen($value > 0)) {  // unexpected
    275. 

  275.                     array_push(
    276. 

  276.                         $tokens1,
    277. 

  277.                         new PHPExcel_Calculation_FormulaToken($value, PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_UNKNOWN)
    278. 

  278.                     );
    279. 

  279.                     $value = "";
    280. 

  280.                 }
    281. 

  281.                 $inString = true;
    282. 

  282.                 $index++;
    283. 

  283.                 continue;
    284. 

  284.              }
    285. 

  285.  
    286. 

  286.             if (substr($this->_formula, $index, 1) == PHPExcel_Calculation_FormulaParser::QUOTE_SINGLE) {
    287. 

  287.                 if (strlen($value) > 0) { // unexpected
    288. 

  288.                     array_push(
    289. 

  289.                         $tokens1,
    290. 

  290.                         new PHPExcel_Calculation_FormulaToken($value, PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_UNKNOWN)
    291. 

  291.                     );
    292. 

  292.                     $value = "";
    293. 

  293.                 }
    294. 

  294.                 $inPath = true;
    295. 

  295.                 $index++;
    296. 

  296.                 continue;
    297. 

  297.             }
    298. 

  298.  
    299. 

  299.             if (substr($this->_formula, $index, 1) == PHPExcel_Calculation_FormulaParser::BRACKET_OPEN) {
    300. 

  300.                 $inRange = true;
    301. 

  301.                 $value .= PHPExcel_Calculation_FormulaParser::BRACKET_OPEN;
    302. 

  302.                 $index++;
    303. 

  303.                 continue;
    304. 

  304.             }
    305. 

  305.  
    306. 

  306.             if (substr($this->_formula, $index, 1) == PHPExcel_Calculation_FormulaParser::ERROR_START) {
    307. 

  307.                 if (strlen($value) > 0) { // unexpected
    308. 

  308.                     array_push(
    309. 

  309.                         $tokens1,
    310. 

  310.                         new PHPExcel_Calculation_FormulaToken($value, PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_UNKNOWN)
    311. 

  311.                     );
    312. 

  312.                     $value = "";
    313. 

  313.                 }
    314. 

  314.                 $inError = true;
    315. 

  315.                 $value .= PHPExcel_Calculation_FormulaParser::ERROR_START;
    316. 

  316.                 $index++;
    317. 

  317.                 continue;
    318. 

  318.             }
    319. 

  319.             
    320. 

  320.             // mark start and end of arrays and array rows
    321. 

  321.             if (substr($this->_formula, $index, 1) == PHPExcel_Calculation_FormulaParser::BRACE_OPEN) {  
    322. 

  322.                 if (strlen($value) > 0) { // unexpected
    323. 

  323.                     array_push(
    324. 

  324.                         $tokens1,
    325. 

  325.                         new PHPExcel_Calculation_FormulaToken($value, PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_UNKNOWN)
    326. 

  326.                     );
    327. 

  327.                     $value = "";
    328. 

  328.                 }
    329. 

  329.  
    330. 

  330.                 $tmp = new PHPExcel_Calculation_FormulaToken("ARRAY", PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_FUNCTION, PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_START);
    331. 

  331.                 array_push($tokens1,     $tmp);
    332. 

  332.                 array_push($stack,         clone $tmp);
    333. 

  333.                 
    334. 

  334.                 $tmp = new PHPExcel_Calculation_FormulaToken("ARRAYROW", PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_FUNCTION, PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_START);
    335. 

  335.                 array_push($tokens1,     $tmp);
    336. 

  336.                 array_push($stack,         clone $tmp);
    337. 

  337.  
    338. 

  338.                 $index++;
    339. 

  339.                 continue;
    340. 

  340.             }
    341. 

  341.  
    342. 

  342.             if (substr($this->_formula, $index, 1) == PHPExcel_Calculation_FormulaParser::SEMICOLON) {  
    343. 

  343.                 if (strlen($value) > 0) {
    344. 

  344.                     array_push(
    345. 

  345.                         $tokens1,
    346. 

  346.                         new PHPExcel_Calculation_FormulaToken($value, PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERAND)
    347. 

  347.                     );
    348. 

  348.                     $value = "";
    349. 

  349.                 }
    350. 

  350.  
    351. 

  351.                 $tmp = array_pop($stack);
    352. 

  352.                 $tmp->setValue("");
    353. 

  353.                 $tmp->setTokenSubType(PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_STOP);
    354. 

  354.                 array_push($tokens1,     $tmp);
    355. 

  355.                 
    356. 

  356.                 $tmp = new PHPExcel_Calculation_FormulaToken(",", PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_ARGUMENT);
    357. 

  357.                 array_push($tokens1,     $tmp);
    358. 

  358.                 
    359. 

  359.                 $tmp = new PHPExcel_Calculation_FormulaToken("ARRAYROW", PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_FUNCTION, PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_START);
    360. 

  360.                 array_push($tokens1,     $tmp);
    361. 

  361.                 array_push($stack,         clone $tmp);
    362. 

  362.                 
    363. 

  363.                 $index++;
    364. 

  364.                 continue;
    365. 

  365.             }
    366. 

  366.  
    367. 

  367.             if (substr($this->_formula, $index, 1) == PHPExcel_Calculation_FormulaParser::BRACE_CLOSE) {  
    368. 

  368.                 if (strlen($value) > 0) {
    369. 

  369.                     array_push(
    370. 

  370.                         $tokens1,
    371. 

  371.                         new PHPExcel_Calculation_FormulaToken($value, PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERAND)
    372. 

  372.                     );
    373. 

  373.                     $value = "";
    374. 

  374.                 }
    375. 

  375.                 
    376. 

  376.                 $tmp = array_pop($stack);
    377. 

  377.                 $tmp->setValue("");
    378. 

  378.                 $tmp->setTokenSubType(PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_STOP);
    379. 

  379.                 array_push($tokens1,     $tmp);
    380. 

  380.                 
    381. 

  381.                 $tmp = array_pop($stack);
    382. 

  382.                 $tmp->setValue("");
    383. 

  383.                 $tmp->setTokenSubType(PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_STOP);
    384. 

  384.                 array_push($tokens1,     $tmp);
    385. 

  385.                 
    386. 

  386.                 $index++;
    387. 

  387.                 continue;
    388. 

  388.             }
    389. 

  389.             
    390. 

  390.             // trim white-space
    391. 

  391.             if (substr($this->_formula, $index, 1) == PHPExcel_Calculation_FormulaParser::WHITESPACE) {
    392. 

  392.                 if (strlen($value) > 0) {
    393. 

  393.                     array_push(
    394. 

  394.                         $tokens1,
    395. 

  395.                         new PHPExcel_Calculation_FormulaToken($value, PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERAND)
    396. 

  396.                     );
    397. 

  397.                     $value = "";
    398. 

  398.                 }
    399. 

  399.                 array_push(
    400. 

  400.                     $tokens1,
    401. 

  401.                     new PHPExcel_Calculation_FormulaToken("", PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_WHITESPACE)
    402. 

  402.                 );
    403. 

  403.                 $index++;
    404. 

  404.                 while ((substr($this->_formula, $index, 1) == PHPExcel_Calculation_FormulaParser::WHITESPACE) && ($index < strlen($this->_formula))) {
    405. 

  405.                     $index++;
    406. 

  406.                 }
    407. 

  407.                 continue;
    408. 

  408.             }
    409. 

  409.             
    410. 

  410.             // multi-character comparators
    411. 

  411.             if (($index + 2) <= strlen($this->_formula)) {
    412. 

  412.                 if (in_array(substr($this->_formula, $index, 2), $COMPARATORS_MULTI)) {
    413. 

  413.                     if (strlen($value) > 0) {
    414. 

  414.                         array_push(
    415. 

  415.                             $tokens1,
    416. 

  416.                             new PHPExcel_Calculation_FormulaToken($value, PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERAND)
