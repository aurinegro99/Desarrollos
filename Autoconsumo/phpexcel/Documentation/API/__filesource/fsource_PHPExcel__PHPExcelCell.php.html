  330.         
    331. 

  331.            // Extract range

    338. 

  338.                list($rangeA, $rangeB) = explode(':', $pRange);

  340.

  341.            // Calculate range outer borders

  344.

  346.            $rangeStart[0]    = PHPExcel_Cell::columnIndexFromString($rangeStart[0]) - 1;

  348.

    356.

  357.         );
