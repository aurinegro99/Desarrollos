  85.     public static function ExcelToPHP($dateValue = 0) {
    86. 

  86.         if (self::$ExcelBaseDate == self::CALENDAR_WINDOWS_1900) {
    87. 

  87.             $myExcelBaseDate = 25569;
    88. 

  88.             //    Adjust for the spurious 29-Feb-1900 (Day 60)
    89. 

  89.             if ($dateValue < 60) {
    90. 

  90.                 $myExcelBaseDate--;
    91. 

  91.             }
    92. 

  92.         } else {
    93. 

  93.             $myExcelBaseDate = 24107;
    94. 

  94.         }
    95. 

  95.  
    96. 

  96.         // Returnvalue
    97. 

  97.         $returnValue = 0;
    98. 

  98.  
    99. 

  99.         // Perform conversion
    100. 

  100.         if ($dateValue >= 1) {
    101. 

  101.             $utcDays = $dateValue - $myExcelBaseDate;
    102. 

  102.             $returnValue = round($utcDays * 24 * 60 * 60);
    103. 

  103.         } else {
    104. 

  104.             $hours = floor($dateValue * 24);
    105. 

  105.             $mins = floor($dateValue * 24 * 60) - $hours * 60;
    106. 

  106.             $secs = floor($dateValue * 24 * 60 * 60) - $hours * 60 * 60 - $mins * 60;
    107. 

  107.             $returnValue = mktime($hours, $mins, $secs);
    108. 

  108.         }
    109. 

  109.  
    110. 

  110.         // Return
    111. 

  111.         return $returnValue;
    112. 

  112.     }

  169.     public static function FormattedPHPToExcel($year, $month, $day, $hours=0, $minutes=0, $seconds=0) {
    170. 

  170.         if (self::$ExcelBaseDate == self::CALENDAR_WINDOWS_1900) {
    171. 

  171.             //
    172. 

  172.             //    Fudge factor for the erroneous fact that the year 1900 is treated as a Leap Year in MS Excel
    173. 

  173.             //    This affects every date following 28th February 1900
    174. 

  174.             //
    175. 

  175.             $excel1900isLeapYear = True;
    176. 

  176.             if (($year == 1900) && ($month <= 2)) { $excel1900isLeapYear = False; }
    177. 

  177.             $myExcelBaseDate = 2415020;
    178. 

  178.         } else {
    179. 

  179.             $myExcelBaseDate = 2416481;
    180. 

  180.             $excel1900isLeapYear = False;
    181. 

  181.         }
    182. 

  182.  
    183. 

  183.         //    Julian base date Adjustment
    184. 

  184.         if ($month > 2) {
    185. 

  185.             $month = $month - 3;
    186. 

  186.         } else {
    187. 

  187.             $month = $month + 9;
    188. 

  188.             $year--;
    189. 

  189.         }
    190. 

  190.  
    191. 

  191.         //    Calculate the Julian Date, then subtract the Excel base date (JD 2415020 = 31-Dec-1899 Giving Excel Date of 0)
    192. 

  192.         $century = substr($year,0,2);
    193. 

  193.         $decade = substr($year,2,2);
    194. 

  194.         $excelDate = floor((146097 * $century) / 4) + floor((1461 * $decade) / 4) + floor((153 * $month + 2) / 5) + $day + 1721119 - $myExcelBaseDate + $excel1900isLeapYear;
    195. 

  195.  
    196. 

  196.         $excelTime = (($hours * 3600) + ($minutes * 60) + $seconds) / 86400;
    197. 

  197.  
    198. 

  198.         return $excelDate + $excelTime;
    199. 

  199.     }
