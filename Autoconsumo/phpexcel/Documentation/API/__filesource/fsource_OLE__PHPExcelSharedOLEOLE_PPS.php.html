    193. 

  193.               . PHPExcel_Shared_OLE::LocalDate2OLE($this->Time2nd)       // 116
    194. 

  194.               . pack("V", isset($this->_StartBlock)? 
    195.

    196. 

  196.               . pack("V", $this->Size)               // 124
    197.
