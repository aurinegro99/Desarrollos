  95.     public function save($pFilename = null) {
    96. 

  96.         // Fetch sheet
    97. 

  97.         $sheet = $this->_phpExcel->getSheet($this->_sheetIndex);
    98. 

  98.         
    99. 

  99.         // Open file
    100. 

  100.         $fileHandle = fopen($pFilename, 'w');
    101. 

  101.         if ($fileHandle === false) {
    102. 

  102.             throw new Exception("Could not open file $pFilename for writing.");
    103. 

  103.         }
    104. 

  104.         
    105. 

  105.         // Get cell collection
    106. 

  106.         $cellCollection = $sheet->getCellCollection();
    107. 

  107.         
    108. 

  108.         // Get column count
    109. 

  109.         $colCount = PHPExcel_Cell::columnIndexFromString($sheet->getHighestColumn());
    110. 

  110.  
    111. 

  111.         // Write headers
    112. 

  112.         $this->_writeHTMLHeader($fileHandle);
    113. 

  113.         $this->_writeStyles($fileHandle, $sheet);
    114. 

  114.         $this->_writeTableHeader($fileHandle);
    115. 

  115.         
    116. 

  116.         // Loop trough cells
    117. 

  117.         $currentRow = -1;
    118. 

  118.         $rowData = array();
    119. 

  119.         foreach ($cellCollection as $cell) {
    120. 

  120.             if ($currentRow != $cell->getRow()) {
    121. 

  121.                 // End previous row?
    122. 

  122.                 if ($currentRow != -1) {
    123. 

  123.                     $this->_writeRow($fileHandle, $sheet, $rowData, ($currentRow - 1));
    124. 

  124.                 }
    125. 

  125.  
    126. 

  126.                 // Set current row
    127. 

  127.                 $currentRow = $cell->getRow();
    128. 

  128.             
    129. 

  129.                 // Start a new row
    130. 

  130.                 $rowData = array();
    131. 

  131.                 for ($i = 0; $i < $colCount; $i++) {
    132. 

  132.                     $rowData[$i] = '';
    133. 

  133.                 }
    134. 

  134.             }
    135. 

  135.                     
    136. 

  136.             // Copy cell
    137. 

  137.             $column = PHPExcel_Cell::columnIndexFromString($cell->getColumn()) - 1;
    138. 

  138.             $rowData[$column] = $cell;
    139. 

  139.         }
    140. 

  140.         
    141. 

  141.         // End last row?
    142. 

  142.         if ($currentRow != -1) {
    143. 

  143.             $this->_writeRow($fileHandle, $sheet, $rowData, ($currentRow - 1));
    144. 

  144.         }
    145. 

  145.                 
    146. 

  146.         // Write footers
    147. 

  147.         $this->_writeTableFooter($fileHandle);
    148. 

  148.         $this->_writeHTMLFooter($fileHandle);
    149. 

  149.         
    150. 

  150.         // Close file
    151. 

  151.         fclose($fileHandle);
    152. 

  152.     }
