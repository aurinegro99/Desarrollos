  91.         $this->_BIG_BLOCK_SIZE  = pow(2,

  100.             if ($this->_FILEH_ == false) {

    102.

    107. 

  107.             }

  109.         // Make an array of PPS's (for Save)

  147.     function _calcSize(&$raList)

  148.     {

  149.         // Calculate Basic Setting
    150. 

  150.         list($iSBDcnt, $iBBcnt, $iPPScnt) = array(0,0,0);
    151. 

  151.         $iSmallLen = 0;
    152. 

  152.         $iSBcnt = 0;
    153. 

  153.         for ($i = 0; $i < count($raList); $i++) {
    154. 

  154.             if ($raList[$i]->Type == PHPExcel_Shared_OLE::OLE_PPS_TYPE_FILE) {
    155. 

  155.                 $raList[$i]->Size = $raList[$i]->_DataLen();
    156. 

  156.                 if ($raList[$i]->Size < PHPExcel_Shared_OLE::OLE_DATA_SIZE_SMALL) {
    157. 

  157.                     $iSBcnt += floor($raList[$i]->Size / $this->_SMALL_BLOCK_SIZE)
    158. 

  158.                                   + (($raList[$i]->Size % $this->_SMALL_BLOCK_SIZE)? 1: 0);
    159. 

  159.                 } else {
    160. 

  160.                     $iBBcnt += (floor($raList[$i]->Size / $this->_BIG_BLOCK_SIZE) +
    161. 

  161.                         (($raList[$i]->Size % $this->_BIG_BLOCK_SIZE)? 1: 0));
    162. 

  162.                 }
    163. 

  163.             }
    164. 

  164.         }
    165. 

  165.         $iSmallLen = $iSBcnt * $this->_SMALL_BLOCK_SIZE;
    166. 

  166.         $iSlCnt = floor($this->_BIG_BLOCK_SIZE / PHPExcel_Shared_OLE::OLE_LONG_INT_SIZE);
    167. 

  167.         $iSBDcnt = floor($iSBcnt / $iSlCnt) + (($iSBcnt % $iSlCnt)? 1:0);
    168. 

  168.         $iBBcnt +=  (floor($iSmallLen / $this->_BIG_BLOCK_SIZE) +
    169. 

  169.                       (( $iSmallLen % $this->_BIG_BLOCK_SIZE)? 1: 0));
    170. 

  170.         $iCnt = count($raList);
    171. 

  171.         $iBdCnt = $this->_BIG_BLOCK_SIZE / PHPExcel_Shared_OLE::OLE_PPS_SIZE;
    172. 

  172.         $iPPScnt = (floor($iCnt/$iBdCnt) + (($iCnt % $iBdCnt)? 1: 0));
    173. 

  173.    
    174. 

  174.         return array($iSBDcnt, $iBBcnt, $iPPScnt);
    175. 

  175.     }

  192.     * Save OLE header

  199.     function _saveHeader($iSBDcnt, $iBBcnt, $iPPScnt)
    200. 

  200.     {
    201. 

  201.         $FILE = $this->_FILEH_;
    202. 

  202.   
    203. 

  203.         // Calculate Basic Setting
    204. 

  204.         $iBlCnt = $this->_BIG_BLOCK_SIZE / PHPExcel_Shared_OLE::OLE_LONG_INT_SIZE;
    205. 

  205.         $i1stBdL = ($this->_BIG_BLOCK_SIZE - 0x4C) / PHPExcel_Shared_OLE::OLE_LONG_INT_SIZE;
    206. 

  206.   
    207. 

  207.         $iBdExL = 0;
    208. 

  208.         $iAll = $iBBcnt + $iPPScnt + $iSBDcnt;
    209. 

  209.         $iAllW = $iAll;
    210. 

  210.         $iBdCntW = floor($iAllW / $iBlCnt) + (($iAllW % $iBlCnt)? 1: 0);
    211. 

  211.         $iBdCnt = floor(($iAll + $iBdCntW) / $iBlCnt) + ((($iAllW+$iBdCntW) % $iBlCnt)? 1: 0);
    212. 

  212.   
    213. 

  213.         // Calculate BD count
    214. 

  214.         if ($iBdCnt > $i1stBdL) {
    215. 

  215.             while (1) {
    216. 

  216.                 $iBdExL++;
    217. 

  217.                 $iAllW++;
    218. 

  218.                 $iBdCntW = floor($iAllW / $iBlCnt) + (($iAllW % $iBlCnt)? 1: 0);
    219. 

  219.                 $iBdCnt = floor(($iAllW + $iBdCntW) / $iBlCnt) + ((($iAllW+$iBdCntW) % $iBlCnt)? 1: 0);
    220. 

  220.                 if ($iBdCnt <= ($iBdExL*$iBlCnt+ $i1stBdL)) {
    221. 

  221.                     break;
    222. 

  222.                 }
    223. 

  223.             }
    224. 

  224.         }
    225. 

  225.   
    226. 

  226.         // Save Header
    227. 

  227.         fwrite($FILE,
    228. 

  228.                   "\xD0\xCF\x11\xE0\xA1\xB1\x1A\xE1"
    229. 

  229.                   . "\x00\x00\x00\x00"
    230. 

  230.                   . "\x00\x00\x00\x00"
    231. 

  231.                   . "\x00\x00\x00\x00"
    232. 

  232.                   . "\x00\x00\x00\x00"
    233. 

  233.                   . pack("v", 0x3b)
    234. 

  234.                   . pack("v", 0x03)
    235. 

  235.                   . pack("v", -2)
    236. 

  236.                   . pack("v", 9)
    237. 

  237.                   . pack("v", 6)
    238. 

  238.                   . pack("v", 0)
    239. 

  239.                   . "\x00\x00\x00\x00"
    240. 

  240.                   . "\x00\x00\x00\x00"
    241. 

  241.                   . pack("V", $iBdCnt) 
    242. 

  242.                   . pack("V", $iBBcnt+$iSBDcnt) //ROOT START
    243. 

  243.                   . pack("V", 0)
    244. 

  244.                   . pack("V", 0x1000)
    245. 

  245.                   . pack("V", 0)                  //Small Block Depot
    246. 

  246.                   . pack("V", 1)
    247. 

  247.           );
    248. 

  248.         // Extra BDList Start, Count
    249. 

  249.         if ($iBdCnt < $i1stBdL) {
    250. 

  250.             fwrite($FILE,
    251. 

  251.                       pack("V", -2).      // Extra BDList Start
    252. 

  252.                       pack("V", 0)        // Extra BDList Count
    253. 

  253.                   );
    254. 

  254.         } else {
    255. 

  255.             fwrite($FILE, pack("V", $iAll+$iBdCnt) . pack("V", $iBdExL));
    256. 

  256.         }
    257. 

  257.  
    258. 

  258.         // BDList
    259. 

  259.         for ($i = 0; $i < $i1stBdL && $i < $iBdCnt; $i++) {
    260. 

  260.             fwrite($FILE, pack("V", $iAll+$i));
    261. 

  261.         }
    262. 

  262.         if ($i < $i1stBdL) {
    263. 

  263.             for ($j = 0; $j < ($i1stBdL-$i); $j++) {
    264. 

  264.                 fwrite($FILE, (pack("V", -1)));
    265. 

  265.             }
    266. 

  266.         }
    267. 

  267.     }
