  282.     public function updateFormulaReferences($pFormula = '', $pBefore = 'A1', $pNumCols = 0, $pNumRows = 0) {    
    283. 

  283.         // Formula stack
    284. 

  284.         $executableFormulaArray = array();
    285. 

  285.         
    286. 

  286.         // Parse formula into a tree of tokens
    287. 

  287.         $objParser = new PHPExcel_Calculation_FormulaParser($pFormula);
    288. 

  288.         
    289. 

  289.         // Loop trough parsed tokens and create an executable formula
    290. 

  290.         $inFunction = false;
    291. 

  291.         $token         = null;
    292. 

  292.         for ($i = 0; $i < $objParser->getTokenCount(); $i++) {
    293. 

  293.             $token = $objParser->getToken($i);
    294. 

  294.  
    295. 

  295.             // Is it a cell reference? Not a cell range?
    296. 

  296.             if ( ($token->getTokenType() == PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERAND) &&
    297. 

  297.                  ($token->getTokenSubType() == PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_RANGE) ) {
    298. 

  298.                      // New cell reference
    299. 

  299.                      $newCellReference = $this->updateCellReference($token->getValue(), $pBefore, $pNumCols, $pNumRows);
    300. 

  300.                      
    301. 

  301.                      // Add adjusted cell coordinate to executable formula array
    302. 

  302.                      array_push($executableFormulaArray, $newCellReference);
    303. 

  303.                          
    304. 

  304.                      continue;
    305. 

  305.             }
    306. 

  306.             
    307. 

  307.             // Is it a subexpression?
    308. 

  308.             if ($token->getTokenType() == PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_SUBEXPRESSION) {
    309. 

  309.                 // Temporary variable
    310. 

  310.                 $tmp = '';
    311. 

  311.                 switch($token->getTokenSubType()) {
    312. 

  312.                     case PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_START:
    313. 

  313.                         $tmp = '(';
    314. 

  314.                         break;
    315. 

  315.                     case PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_STOP:
    316. 

  316.                         $tmp = ')';
    317. 

  317.                         break;
    318. 

  318.                 }
    319. 

  319.                     
    320. 

  320.                 // Add to executable formula array
    321. 

  321.                 array_push($executableFormulaArray, $tmp);
    322. 

  322.                          
    323. 

  323.                 continue;
    324. 

  324.             }
    325. 

  325.                 
    326. 

  326.             // Is it a function?
    327. 

  327.             if ($token->getTokenType() == PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_FUNCTION) {
    328. 

  328.                 // Temporary variable
    329. 

  329.                 $tmp = '';
    330. 

  330.                     
    331. 

  331.                 // Check the function type
    332. 

  332.                 if ($token->getValue() == 'ARRAY' || $token->getValue() == 'ARRAYROW') {
    333. 

  333.                     // An array or an array row...
    334. 

  334.                     $tmp = '(';
    335. 

  335.                 } else {
    336. 

  336.                     // A regular function call...
    337. 

  337.                     switch($token->getTokenSubType()) {
    338. 

  338.                         case PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_START:
    339. 

  339.                             $tmp = strtoupper($token->getValue()) . '(';
    340. 

  340.                             $inFunction = true;
    341. 

  341.                             break;
    342. 

  342.                         case PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_STOP:
    343. 

  343.                             $tmp = ')';
    344. 

  344.                             break;
    345. 

  345.                     }
    346. 

  346.                 }
    347. 

  347.                     
    348. 

  348.                 // Add to executable formula array
    349. 

  349.                 array_push($executableFormulaArray, $tmp);
    350. 

  350.                      
    351. 

  351.                 continue;
    352. 

  352.             }
    353. 

  353.                 
    354. 

  354.             // Is it text?
    355. 

  355.             if ( ($token->getTokenType() == PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERAND) &&
    356. 

  356.                  ($token->getTokenSubType() == PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_TEXT) ) {
    357. 

  357.                      // Add to executable formula array
    358. 

  358.                      array_push($executableFormulaArray, '"' . $token->getValue() . '"');
    359. 

  359.                          
    360. 

  360.                      continue;
    361. 

  361.             }
    362. 

  362.                 
    363. 

  363.             // Is it something else?
    364. 

  364.             array_push($executableFormulaArray, $token->getValue());
    365. 

  365.         }
    366. 

  366.         
    367. 

  367.         // Return result
    368. 

  368.         return '=' . implode(' ', $executableFormulaArray);
    369. 

  369.     }

    403. 

  403.         if (eregi(':', $pCellRange)) {
    404. 

  404.             // Range
    405. 

  405.             $referenceA = '';
    406.
