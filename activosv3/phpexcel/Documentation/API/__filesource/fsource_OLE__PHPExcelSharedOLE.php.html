  103.     function read($file)
    104. 

  104.     {
    105. 

  105.         $fh = @fopen($file, "r");
    106. 

  106.         if (!$fh) {
    107. 

  107.             throw new Exception("Can't open file $file");
    108. 

  108.         }
    109. 

  109.         $this->_file_handle = $fh;
    110. 

  110.  
    111. 

  111.         $signature = fread($fh, 8);
    112. 

  112.         if ("\xD0\xCF\x11\xE0\xA1\xB1\x1A\xE1" != $signature) {
    113. 

  113.             throw new Exception("File doesn't seem to be an OLE container.");
    114. 

  114.         }
    115. 

  115.         fseek($fh, 28);
    116. 

  116.         if (fread($fh, 2) != "\xFE\xFF") {
    117. 

  117.             // This shouldn't be a problem in practice
    118. 

  118.             throw new Exception("Only Little-Endian encoding is supported.");
    119. 

  119.         }
    120. 

  120.         // Size of blocks and short blocks in bytes
    121. 

  121.         $this->bigBlockSize = pow(2, $this->_readInt2($fh));
    122. 

  122.         $this->smallBlockSize  = pow(2, $this->_readInt2($fh));
    123. 

  123.  
    124. 

  124.         // Skip UID, revision number and version number
    125. 

  125.         fseek($fh, 44);
    126. 

  126.         // Number of blocks in Big Block Allocation Table
    127. 

  127.         $bbatBlockCount = $this->_readInt4($fh);
    128. 

  128.  
    129. 

  129.         // Root chain 1st block
    130. 

  130.         $directoryFirstBlockId = $this->_readInt4($fh);
    131. 

  131.  
    132. 

  132.         // Skip unused bytes
    133. 

  133.         fseek($fh, 56);
    134. 

  134.         // Streams shorter than this are stored using small blocks
    135. 

  135.         $this->bigBlockThreshold = $this->_readInt4($fh);
    136. 

  136.         // Block id of first sector in Short Block Allocation Table
    137. 

  137.         $sbatFirstBlockId = $this->_readInt4($fh);
    138. 

  138.         // Number of blocks in Short Block Allocation Table
    139. 

  139.         $sbbatBlockCount = $this->_readInt4($fh);
    140. 

  140.         // Block id of first sector in Master Block Allocation Table
    141. 

  141.         $mbatFirstBlockId = $this->_readInt4($fh);
    142. 

  142.         // Number of blocks in Master Block Allocation Table
    143. 

  143.         $mbbatBlockCount = $this->_readInt4($fh);
    144. 

  144.         $this->bbat = array();
    145. 

  145.  
    146. 

  146.         // Remaining 4 * 109 bytes of current block is beginning of Master
    147. 

  147.         // Block Allocation Table
    148. 

  148.         $mbatBlocks = array();
    149. 

  149.         for ($i = 0; $i < 109; $i++) {
    150. 

  150.             $mbatBlocks[] = $this->_readInt4($fh);
    151. 

  151.         }
    152. 

  152.  
    153. 

  153.         // Read rest of Master Block Allocation Table (if any is left)
    154. 

  154.         $pos = $this->_getBlockOffset($mbatFirstBlockId);
    155. 

  155.         for ($i = 0; $i < $mbbatBlockCount; $i++) {
    156. 

  156.             fseek($fh, $pos);
    157. 

  157.             for ($j = 0; $j < $this->bigBlockSize / 4 - 1; $j++) {
    158. 

  158.                 $mbatBlocks[] = $this->_readInt4($fh);
    159. 

  159.             }
    160. 

  160.             // Last block id in each block points to next block
    161. 

  161.             $pos = $this->_getBlockOffset($this->_readInt4($fh));
    162. 

  162.         }
    163. 

  163.  
    164. 

  164.         // Read Big Block Allocation Table according to chain specified by
    165. 

  165.         // $mbatBlocks
    166. 

  166.         for ($i = 0; $i < $bbatBlockCount; $i++) {
    167. 

  167.             $pos = $this->_getBlockOffset($mbatBlocks[$i]);
    168. 

  168.             fseek($fh, $pos);
    169. 

  169.             for ($j = 0 ; $j < $this->bigBlockSize / 4; $j++) {
    170. 

  170.                 $this->bbat[] = $this->_readInt4($fh);
    171. 

  171.             }
    172. 

  172.         }
    173. 

  173.  
    174. 

  174.         // Read short block allocation table (SBAT)
    175. 

  175.         $this->sbat = array();
    176. 

  176.         $shortBlockCount = $sbbatBlockCount * $this->bigBlockSize / 4;
    177. 

  177.         $sbatFh = $this->getStream($sbatFirstBlockId);
    178. 

  178.         for ($blockId = 0; $blockId < $shortBlockCount; $blockId++) {
    179. 

  179.             $this->sbat[$blockId] = $this->_readInt4($sbatFh);
    180. 

  180.         }
    181. 

  181.         fclose($sbatFh);
    182. 

  182.  
    183. 

  183.         $this->_readPpsWks($directoryFirstBlockId);
    184. 

  184.  
    185. 

  185.         return true;
    186. 

  186.     }

  274.     function _readPpsWks($blockId)
    275. 

  275.     {
    276. 

  276.         $fh = $this->getStream($blockId);
    277. 

  277.         for ($pos = 0; ; $pos += 128) {
    278. 

  278.             fseek($fh, $pos, SEEK_SET);
    279. 

  279.             $nameUtf16 = fread($fh, 64);
    280. 

  280.             $nameLength = $this->_readInt2($fh);
    281. 

  281.             $nameUtf16 = substr($nameUtf16, 0, $nameLength - 2);
    282. 

  282.             // Simple conversion from UTF-16LE to ISO-8859-1
    283. 

  283.             $name = str_replace("\x00", "", $nameUtf16);
    284. 

  284.             $type = $this->_readInt1($fh);
    285. 

  285.             switch ($type) {
    286. 

  286.             case self::OLE_PPS_TYPE_ROOT:
    287. 

  287.                 $pps = new PHPExcel_Shared_OLE_PPS_Root(null, null, array());
    288. 

  288.                 $this->root = $pps;
    289. 

  289.                 break;
    290. 

  290.             case self::OLE_PPS_TYPE_DIR:
    291. 

  291.                 $pps = new PHPExcel_Shared_OLE_PPS(null, null, null, null, null,
    292. 

  292.                                    null, null, null, null, array());
    293. 

  293.                 break;
    294. 

  294.             case self::OLE_PPS_TYPE_FILE:
    295. 

  295.                 $pps = new PHPExcel_Shared_OLE_PPS_File($name);
    296. 

  296.                 break;
    297. 

  297.             default:
    298. 

  298.                 continue;
    299. 

  299.             }
    300. 

  300.             fseek($fh, 1, SEEK_CUR);
    301. 

  301.             $pps->Type    = $type;
    302. 

  302.             $pps->Name    = $name;
    303. 

  303.             $pps->PrevPps = $this->_readInt4($fh);
    304. 

  304.             $pps->NextPps = $this->_readInt4($fh);
    305. 

  305.             $pps->DirPps  = $this->_readInt4($fh);
    306. 

  306.             fseek($fh, 20, SEEK_CUR);
    307. 

  307.             $pps->Time1st = self::OLE2LocalDate(fread($fh, 8));
    308. 

  308.             $pps->Time2nd = self::OLE2LocalDate(fread($fh, 8));
    309. 

  309.             $pps->_StartBlock = $this->_readInt4($fh);
    310. 

  310.             $pps->Size = $this->_readInt4($fh);
    311. 

  311.             $pps->No = count($this->_list);
    312. 

  312.             $this->_list[] = $pps;
    313. 

  313.  
    314. 

  314.             // check if the PPS tree (starting from root) is complete
    315. 

  315.             if (isset($this->root) &&
    316. 

  316.                 $this->_ppsTreeComplete($this->root->No)) {
    317. 

  317.  
    318. 

  318.                 break;
    319. 

  319.             }
    320. 

  320.         }
    321. 

  321.         fclose($fh);
    322. 

  322.  
    323. 

  323.         // Initialize $pps->children on directories
    324. 

  324.         foreach ($this->_list as $pps) {
    325. 

  325.             if ($pps->Type == self::OLE_PPS_TYPE_DIR || $pps->Type == self::OLE_PPS_TYPE_ROOT) {
    326. 

  326.                 $nos = array($pps->DirPps);
    327. 

  327.                 $pps->children = array();
    328. 

  328.                 while ($nos) {
    329. 

  329.                     $no = array_pop($nos);
    330. 

  330.                     if ($no != -1) {
    331. 

  331.                         $childPps = $this->_list[$no];
    332. 

  332.                         $nos[] = $childPps->PrevPps;
    333. 

  333.                         $nos[] = $childPps->NextPps;
    334. 

  334.                         $pps->children[] = $childPps;
    335. 

  335.                     }
    336. 

  336.                 }
    337. 

  337.             }
    338. 

  338.         }
    339. 

  339.  
    340. 

  340.         return true;
    341. 

  341.     }
